MNQ futures study testing whether the final daily high or low forms late in the session, using mutually exclusive close windows and regime-based conditioning.

Executive Finding

15:00–16:00 EST is the primary late-day auction-completion window.

The dominant signal is not a blind breakout trigger. The research is better framed as a risk-regime model: identify when the market is more likely to complete a new daily extreme late, then manage continuation, exhaustion, and rejection risk accordingly.

IB extension is the strongest variable identified so far.

The cleanest continuation zone appears near 1.25x to 2.00x IB extension. Above 2.00x IB, the response weakens, which may indicate exhaustion or that the day auction completed earlier.

Python Study Code
from pathlib import Path

script = Path("/mnt/data/compute_mutually_exclusive_close_studies.py")

content = r'''"""
MNQ Close Studies — mutually exclusive outcome buckets.

Definitions:
- Session/day high-low: 09:30–16:15 EST
- IB: 09:30–10:30 EST
- Mutually exclusive outcome buckets:
  * Close2h-only: 14:00–15:00
  * Close1h-only: 15:00–16:00
  * Last15-only: 16:00–16:15

Event timing uses first occurrence timestamp of day high/low.
Rolling quintiles use prior 20 trading days only.
"""

import pandas as pd
import numpy as np
import math
from datetime import time

DATA_PATH = "MNQ LT DATA.csv"

def mask_between(t_series, start_t, end_t):
    return (t_series >= start_t) & (t_series < end_t)

def time_in(series_dt, start_t, end_t):
    tt = pd.to_datetime(series_dt).dt.time
    return (tt >= start_t) & (tt < end_t)

def wilson_ci(k, n, z=1.959963984540054):
    if n == 0:
        return (np.nan, np.nan)
    p = k / n
    denom = 1 + z*z/n
    center = (p + z*z/(2*n)) / denom
    half = z * math.sqrt((p*(1-p) + z*z/(4*n))/n) / denom
    return center-half, center+half

cols = pd.read_csv(DATA_PATH, nrows=0).columns.tolist()

dt_candidates = [c for c in cols if any(k in c.lower() for k in ["date time","datetime","timestamp"])]
if not dt_candidates:
    dt_candidates = [c for c in cols if ("date" in c.lower() and "time" in c.lower())]

dt_col = dt_candidates[0]
high_col = "High" if "High" in cols else ("HIGH" if "HIGH" in cols else "high")
low_col  = "Low" if "Low" in cols else ("LOW" if "LOW" in cols else "low")

df = pd.read_csv(DATA_PATH, usecols=[dt_col, high_col, low_col], low_memory=False)
df["dt"] = pd.to_datetime(df[dt_col], errors="coerce")
df = df.dropna(subset=["dt"]).copy()
df = df.rename(columns={high_col:"High", low_col:"Low"})
df["date"] = df["dt"].dt.date

tser = df["dt"].dt.time

IB_START = time(9,30)
IB_END   = time(10,30)
SESS_START = time(9,30)
SESS_END   = time(16,15)

WINDOWS = {
    "close2h_only": ("14:00–15:00", time(14,0), time(15,0)),
    "close1h_only": ("15:00–16:00", time(15,0), time(16,0)),
    "last15_only":  ("16:00–16:15", time(16,0), time(16,15)),
}

df["_in_sess"] = mask_between(tser, SESS_START, SESS_END)
sess = df[df["_in_sess"]].copy()

day_agg = sess.groupby("date").agg(day_high=("High","max"), day_low=("Low","min")).reset_index()

sess_hi = sess.merge(day_agg[["date","day_high"]], on="date", how="inner")
hi_first = (sess_hi.loc[sess_hi["High"] == sess_hi["day_high"]]
    .groupby("date")["dt"].min().rename("day_high_first_dt").reset_index())

sess_lo = sess.merge(day_agg[["date","day_low"]], on="date", how="inner")
lo_first = (sess_lo.loc[sess_lo["Low"] == sess_lo["day_low"]]
    .groupby("date")["dt"].min().rename("day_low_first_dt").reset_index())

occ = day_agg.merge(hi_first, on="date", how="inner").merge(lo_first, on="date", how="inner")

rows = []
N = len(occ)

for key, (label, start_t, end_t) in WINDOWS.items():
    occ[f"high_in_{key}"] = time_in(occ["day_high_first_dt"], start_t, end_t)
    occ[f"low_in_{key}"] = time_in(occ["day_low_first_dt"], start_t, end_t)
    occ[f"event_{key}"] = occ[f"high_in_{key}"] | occ[f"low_in_{key}"]

    k = int(occ[f"event_{key}"].sum())
    p = k / N
    ci_lo, ci_hi = wilson_ci(k, N)

    rows.append({
        "window": label,
        "N_days": N,
        "k_event_true": k,
        "probability": p,
        "ci95_low": ci_lo,
        "ci95_high": ci_hi
    })

metrics = pd.DataFrame(rows)
print(metrics)
metrics.to_csv("study1_unconditional_mutually_exclusive_metrics.csv", index=False)
occ.to_csv("daily_occurrences_mutually_exclusive.csv", index=False)

print("\\nFiles exported:")
print("- study1_unconditional_mutually_exclusive_metrics.csv")
print("- daily_occurrences_mutually_exclusive.csv")
'''

script.write_text(content, encoding="utf-8")
print(f"Saved script to: {script}")
